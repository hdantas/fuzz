 87          chunked_label = "" 
 88          cursor        = 0

 90          while cursor < len(self.label):

 93              # if the end of the current chunk contains a backslash or double-quote, back off some. 
 94              if cursor + amount < len(self.label): 
 95                  while self.label[cursor+amount] == '\\' or self.label[cursor+amount] == '"': 
 96                      amount -= 1 
 97   
 98              chunked_label += self.label[cursor:cursor+amount] + "\\\n" 
 99              cursor        += amount 
100   
101          # if node width and height were not explicitly specified, make a best effort guess to create something nice. 
102          if not self.gml_width:
